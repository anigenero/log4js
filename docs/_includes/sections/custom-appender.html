Custom Appenders

You can create custom log appenders for specific use cases.

{% highlight typescript %} @Appender('MyAppender') export class MyAppender extends LogAppender { public static get appenderName(): string { return 'MyAppender'; } public append(logEvent: ILogEvent) { if (logEvent.level <= this.getLogLevel()) { const message = this.format(logEvent); // handle message } } } {% endhighlight %} {% highlight typescript %} import { MyAppender } from './myappender.ts'; configure({ // ... appenders: [{ appender: MyAppender, // can also refer by name ('MyAppender') config: { // ... } }] }); {% endhighlight %}
